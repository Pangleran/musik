Puppeteer

Library Node.js untuk mengontrol browser Chrome/Chromium secara programatik — scraping, testing, otomasi, screenshot, dan PDF generation.

Node.js v18+ Chromium v22.x Latest

01 Pengantar

Puppeteer adalah library resmi Google untuk mengontrol Chrome/Chromium melalui DevTools Protocol. Cocok untuk automasi browser, web scraping, pengujian UI, dan pembuatan screenshot/PDF.

🕷️

Web Scraping

Ambil konten dari halaman web yang membutuhkan JavaScript rendering.

🧪

Testing UI

Uji aplikasi web secara end-to-end dengan simulasi klik dan input pengguna.

📸

Screenshot

Tangkap halaman penuh atau elemen tertentu sebagai gambar PNG/JPEG.

📄

PDF Generator

Buat PDF dari halaman web dengan kontrol penuh atas ukuran dan margin.

ℹ️ Puppeteer vs Playwright

Puppeteer fokus pada Chrome/Chromium. Jika butuh Firefox/Safari juga, pertimbangkan Playwright. Keduanya memiliki API yang mirip.

02 Instalasi

Install via npm

bash
# Install puppeteer (sudah include Chromium ~170MB)
npm install puppeteer

# Atau hanya library tanpa download Chromium
npm install puppeteer-core

Di Linux, Chromium membutuhkan beberapa library sistem. Jalankan: npx puppeteer browsers install chrome untuk memperbaiki dependensi otomatis.

(async () => {

11 Waiting Strategies

Salah satu kunci sukses scraping adalah strategi menunggu yang tepat. Hindari setTimeout — gunakan metode berikut.

javascript
// Tunggu selector muncul di DOM
await page.waitForSelector('.result');

// Tunggu dengan opsi
await page.waitForSelector('#content', {
  visible: true,     // Elemen harus terlihat
  hidden: false,      // Tunggu hingga disembunyikan
  timeout: 5000,      // Timeout (ms)
});

// Tunggu navigasi selesai
await page.waitForNavigation({ waitUntil: 'networkidle2' });

// Tunggu dan klik sekaligus (race condition safe)
const [response] = await Promise.all([
  page.waitForNavigation(),
  page.click('a.nav-link'),
]);

// Tunggu kondisi fungsi bernilai truthy
await page.waitForFunction(() =>
  document.querySelector('#loading')?.style.display === 'none'
);

// Tunggu waktu (gunakan seminimal mungkin)
await page.waitForTimeout(1000); // deprecated, gunakan setTimeout
await new Promise(r => setTimeout(r, 1000));

// Tunggu request jaringan spesifik
await page.waitForRequest('https://api.example.com/data');
await page.waitForResponse(res => res.url().includes('/api/'));
💡 Tips: Pola Aman untuk SPA

Untuk Single Page Application (React/Vue/Angular), gunakan waitForSelector + networkidle2 bersama-sama untuk memastikan data sudah di-render.

17 Concurrency

Scrape Banyak URL Paralel

javascript
const puppeteer = require('puppeteer');

async function scrapeUrl(browser, url) {
  const page = await browser.newPage();
  try {
    await page.goto(url, { waitUntil: 'domcontentloaded' });
    const data = await page.evaluate(() => ({
      title: document.title,
      text: document.body?.innerText?.slice(0, 200),
    }));
    return { url, ...data };
  } finally {
    await page.close(); // Selalu tutup page!
  }
}

async function scrapeAll(urls, concurrency = 5) {
  const browser = await puppeteer.launch();
  const results = [];

  // Proses batch untuk membatasi concurrency
  for (let i = 0; i < urls.length; i += concurrency) {
    const batch = urls.slice(i, i + concurrency);
    const batchResults = await Promise.all(
      batch.map(url => scrapeUrl(browser, url))
    );
    results.push(...batchResults);
  }

  await browser.close();
  return results;
}

// Contoh penggunaan
const urls = [
  'https://example.com/page/1',
  'https://example.com/page/2',
  'https://example.com/page/3',
];

scrapeAll(urls, 3).then(console.log);

19 Error Handling

javascript
const puppeteer = require('puppeteer');
const { TimeoutError } = require('puppeteer');

async function safeScrape(url) {
  let browser;
  try {
    browser = await puppeteer.launch();
    const page = await browser.newPage();

    // Set timeout global untuk semua operasi
    page.setDefaultTimeout(15000);
    page.setDefaultNavigationTimeout(30000);

    await page.goto(url);
    const data = await page.$eval('h1', el => el.textContent);
    return data;

  } catch (err) {
    if (err instanceof TimeoutError) {
      console.error('Timeout! Halaman tidak merespons.');
    } else if (err.message.includes('net::ERR_')) {
      console.error('Network error:', err.message);
    } else {
      console.error('Error tidak terduga:', err);
    }
    return null;

  } finally {
    // Browser SELALU ditutup, bahkan saat error
    if (browser) await browser.close();
  }
}

Scroll Infinite Loading

javascript
async function autoScroll(page) {
  await page.evaluate(async () => {
    await new Promise((resolve) => {
      let totalHeight = 0;
      const distance = 300;
      const timer = setInterval(() => {
        window.scrollBy(0, distance);
        totalHeight += distance;
        if (totalHeight >= document.body.scrollHeight) {
          clearInterval(timer);
          resolve();
        }
      }, 200);
    });
  });
}

javascript
const puppeteer = require('puppeteer');
const fs = require('fs');

(async () => {
  const browser = await puppeteer.launch({ headless: 'new' });
  const page = await browser.newPage();

  // Konfigurasi
  await page.setViewport({ width: 1280, height: 800 });
  await page.setUserAgent('Mozilla/5.0 ...');
  page.setDefaultTimeout(20000);

  const allData = [];
  let page_num = 1;

  while (true) {
    await page.goto(`https://example.com/products?page=${page_num}`);

    // Tunggu konten
    await page.waitForSelector('.product-list');

    // Ambil data
    const products = await page.$$eval('.product-card', cards =>
      cards.map(card => ({
        name: card.querySelector('.name')?.textContent?.trim(),
        price: card.querySelector('.price')?.textContent?.trim(),
        url: card.querySelector('a')?.href,
      }))
    );

    if (products.length === 0) break;
    allData.push(...products);

    // Cek tombol next
    const nextBtn = await page.$('.pagination-next:not(.disabled)');
    if (!nextBtn) break;

    page_num++;
  }

  // Simpan hasil
  fs.writeFileSync('output.json', JSON.stringify(allData, null, 2));
  console.log(`✅ Berhasil scrape ${allData.length} produk`);

  await browser.close();
})();
💡 Best Practice

Selalu gunakan blok try/finally untuk memastikan browser.close() selalu dipanggil. Gunakan Promise.all() saat menggabungkan aksi dengan waitForNavigation. Hindari page.waitForTimeout() — gunakan selector atau fungsi sebagai kondisi tunggu.
